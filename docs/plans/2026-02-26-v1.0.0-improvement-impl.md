```go
package dashboard

import (
	"context"
	"database/sql"

	"github.com/uptrace/bun"
)

type Repository struct {
	db *bun.DB
}

func NewRepository(db *bun.DB) *Repository {
	return &Repository{db: db}
}

func (r *Repository) GetStats(ctx context.Context, siteSchema string) (*DashboardStats, error) {
	stats := &DashboardStats{}

	// Posts 统计 - 使用 bun.Prepend 防止 SQL 注入
	postsQuery := "SELECT COALESCE(status, 0), COUNT(*) FROM " + bun.Ident(siteSchema) + ".sfc_site_posts WHERE deleted_at IS NULL GROUP BY status"
	rows, err := r.db.QueryContext(ctx, postsQuery)
	if err != nil && err != sql.ErrNoRows {
		return nil, err
	}
	defer rows.Close()
	for rows.Next() {
		var status int
		var count int64
		if err := rows.Scan(&status, &count); err != nil {
			return nil, err
		}
		stats.Posts.Total += count
		switch status {
		case 1: // published
			stats.Posts.Published = count
		case 2: // draft
			stats.Posts.Draft = count
		case 3: // scheduled
			stats.Posts.Scheduled = count
		}
	}

	// Users 统计
	stats.Users.Total = 1 // 超级用户

	// Comments 统计
	commentsQuery := "SELECT COALESCE(status, 0), COUNT(*) FROM " + bun.Ident(siteSchema) + ".sfc_site_comments WHERE deleted_at IS NULL GROUP BY status"
	commentRows, err := r.db.QueryContext(ctx, commentsQuery)
	if err != nil && err != sql.ErrNoRows {
		return nil, err
	}
	defer commentRows.Close()
	for commentRows.Next() {
		var status int
		var count int64
		if err := commentRows.Scan(&status, &count); err != nil {
			return nil, err
		}
		stats.Comments.Total += count
		switch status {
		case 1: // pending
			stats.Comments.Pending = count
		case 2: // approved
			stats.Comments.Approved = count
		case 3: // spam
			stats.Comments.Spam = count
		}
	}

	// Media 统计
	mediaQuery := "SELECT COUNT(*), COALESCE(SUM(file_size), 0) FROM " + bun.Ident(siteSchema) + ".sfc_site_media_files WHERE deleted_at IS NULL"
	if err := r.db.QueryContext(ctx, mediaQuery).Scan(&stats.Media.Total, &stats.Media.StorageUsed); err != nil && err != sql.ErrNoRows {
		return nil, err
	}

	return stats, nil
}
```

```typescript
import { queryClient } from './api-client';

export interface DashboardStats {
  posts: {
    total: number;
    published: number;
    draft: number;
    scheduled: number;
  };
  users: {
    total: number;
    active: number;
    inactive: number;
  };
  comments: {
    total: number;
    pending: number;
    approved: number;
    spam: number;
  };
  media: {
    total: number;
    storage_used: number;
  };
}

export function formatBytes(bytes: number): string {
  if (bytes === 0) return '0 B';
  const k = 1024;
  const sizes = ['B', 'KB', 'MB', 'GB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
}

export async function fetchDashboardStats(): Promise<DashboardStats> {
  const response = await fetch('/api/v1/site/dashboard/stats');
  if (!response.ok) {
    throw new Error('Failed to fetch dashboard stats');
  }
  const data = await response.json();
  return data.data;
}

export const dashboardKeys = {
  stats: ['dashboard', 'stats'] as const,
};
```

```tsx
import { useState, useEffect } from 'react';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { Button } from '@/components/ui/button';
import { Command, CommandEmpty, CommandGroup, CommandInput, CommandItem, CommandList } from '@/components/ui/command';
import { ChevronDown, Check } from 'lucide-react';
import { categoriesApi, Category } from '@/lib/content-api';

interface CategorySelectProps {
  value: string[];
  onChange: (values: string[]) => void;
  siteSlug: string;
}

export function CategorySelect({ value, onChange, siteSlug }: CategorySelectProps) {
  const [open, setOpen] = useState(false);
  const [categories, setCategories] = useState<Category[]>([]);
  const [search, setSearch] = useState('');

  useEffect(() => {
    categoriesApi.list(siteSlug).then(setCategories).catch(console.error);
  }, [siteSlug]);

  const selectedLabels = categories
    .filter((c) => value.includes(c.id))
    .map((c) => c.name)
    .join(', ');

  const handleSelect = (id: string) => {
    if (value.includes(id)) {
      onChange(value.filter((v) => v !== id));
    } else {
      onChange([...value, id]);
    }
  };

  return (
    <Popover open={open} onOpenChange={setOpen}>
      <PopoverTrigger asChild>
        <Button variant="outline" className="w-full justify-start">
          {selectedLabels || '选择分类...'}
          <ChevronDown className="ml-auto h-4 w-4" />
        </Button>
      </PopoverTrigger>
      <PopoverContent className="p-0" align="start">
        <Command>
          <CommandInput placeholder="搜索分类..." value={search} onValueChange={setSearch} />
          <CommandList>
            <CommandEmpty>未找到分类</CommandEmpty>
            <CommandGroup>
              {categories.map((category) => (
                <CommandItem key={category.id} value={category.name} onSelect={() => handleSelect(category.id)}>
                  <Check className={`mr-2 h-4 w-4 ${value.includes(category.id) ? 'opacity-100' : 'opacity-0'}`} />
                  {category.name}
                </CommandItem>
              ))}
            </CommandGroup>
          </CommandList>
        </Command>
      </PopoverContent>
    </Popover>
  );
}
```

**Step 3: 创建 TagSelect 组件**

```tsx
import { useState, useEffect } from 'react';
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
import { Button } from '@/components/ui/button';
import { Command, CommandEmpty, CommandGroup, CommandInput, CommandItem, CommandList } from '@/components/ui/command';
import { Badge } from '@/components/ui/badge';
import { ChevronDown, X, Plus } from 'lucide-react';
import { tagsApi, Tag } from '@/lib/content-api';

interface TagSelectProps {
  value: string[];
  onChange: (values: string[]) => void;
  siteSlug: string;
}

export function TagSelect({ value, onChange, siteSlug }: TagSelectProps) {
  const [open, setOpen] = useState(false);
  const [tags, setTags] = useState<Tag[]>([]);
  const [search, setSearch] = useState('');

  useEffect(() => {
    tagsApi.list(siteSlug).then(setTags).catch(console.error);
  }, [siteSlug]);

  const selectedTags = tags.filter((t) => value.includes(t.id));

  const handleSelect = (id: string) => {
    if (value.includes(id)) {
      onChange(value.filter((v) => v !== id));
    } else {
      onChange([...value, id]);
    }
  };

  const handleCreate = async () => {
    if (!search.trim()) return;
    try {
      const newTag = await tagsApi.create(siteSlug, { name: search.trim(), slug: search.trim().toLowerCase() });
      onChange([...value, newTag.id]);
      setSearch('');
    } catch (error) {
      console.error('Failed to create tag:', error);
    }
  };

  return (
    <div className="space-y-2">
      <Popover open={open} onOpenChange={setOpen}>
        <PopoverTrigger asChild>
          <Button variant="outline" className="w-full justify-start">
            <Plus className="mr-2 h-4 w-4" />
            添加标签...
            <ChevronDown className="ml-auto h-4 w-4" />
          </Button>
        </PopoverTrigger>
        <PopoverContent className="p-0" align="start">
          <Command>
            <CommandInput placeholder="搜索标签..." value={search} onValueChange={setSearch} />
            <CommandList>
              <CommandEmpty>
                <Button variant="ghost" size="sm" onClick={handleCreate}>
                  创建 "{search}"
                </Button>
              </CommandEmpty>
              <CommandGroup>
                {tags.map((tag) => (
                  <CommandItem key={tag.id} value={tag.name} onSelect={() => handleSelect(tag.id)}>
                    <Check className={`mr-2 h-4 w-4 ${value.includes(tag.id) ? 'opacity-100' : 'opacity-0'}`} />
                    {tag.name}
                  </CommandItem>
                ))}
              </CommandGroup>
            </CommandList>
          </Command>
        </PopoverContent>
      </Popover>

      {selectedTags.length > 0 && (
        <div className="flex flex-wrap gap-1">
          {selectedTags.map((tag) => (
            <Badge key={tag.id} variant="secondary" className="gap-1">
              {tag.name}
              <X className="h-3 w-3 cursor-pointer" onClick={() => handleSelect(tag.id)} />
            </Badge>
          ))}
        </div>
      )}
    </div>
  );
}
```

```tsx
interface SettingsFormProps {
  siteSlug: string;
}

export function SettingsForm({ siteSlug }: SettingsFormProps) {
  const [formData, setFormData] = useState<Record<string, string>>({});
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [saving, setSaving] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setErrors({});
    setSaving(true);

    try {
      await settingsApi.update(siteSlug, formData);
      toast.success('设置已保存');
    } catch (error: unknown) {
      if (error && typeof error === 'object' && 'response' in error) {
        const axiosError = error as { response?: { data?: { error?: string } } };
        if (axiosError.response?.data?.error) {
          const errorMsg = axiosError.response.data.error;
          if (errorMsg.includes('site_name')) {
            setErrors({ site_name: errorMsg });
          } else if (errorMsg.includes('site_url')) {
            setErrors({ site_url: errorMsg });
          } else {
            toast.error(errorMsg);
          }
        }
      } else {
        toast.error('保存失败，请重试');
      }
    } finally {
      setSaving(false);
    }
  };

  // 在每个字段后面添加错误显示：
  // {errors.field_name && <p className="text-sm text-red-500">{errors.field_name}</p>}
}
```
